# Example manual

```shell
# Install typescript
npm install -D typescript

# Install Linter
npm install -D \
  eslint \
  eslint-config-prettier \
  eslint-plugin-import \
  eslint-plugin-jest \
  eslint-plugin-prettier \
  @typescript-eslint/eslint-plugin \
  @typescript-eslint/parser \
  prettier \
  eslint-plugin-react \
  eslint-plugin-react-hooks \
  eslint-plugin-jsx-a11y \
  stylelint \
  stylelint-config-prettier \
  stylelint-config-standard \
  stylelint-order \
  prettier-stylelint

wget https://github.com/bobstrange/js-sandbox/blob/master/typescript/ts-boilerplate/.eslintrc.js
wget https://github.com/bobstrange/js-sandbox/blob/master/typescript/ts-boilerplate/.prettierrc.js
wget https://github.com/bobstrange/js-sandbox/blob/master/typescript/ts-boilerplate/.stylelintrc.js

# Install react
npm install --save react react-dom
npm install -D @types/react @types/react-dom

```
